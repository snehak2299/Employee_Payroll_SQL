*****UC2*****
 create table employee_payroll (          // create table
    -> id INT unsigned NOT NULL AUTO_INCREMENT,
    -> name VARCHAR(150) NOT NULL,
    -> salary Double NOT NULL,
    -> start DATE NOT NULL,
    -> PRIMARY KEY (id)
    -> );


mysql> DESCRIBE employee_payroll;     // learn data type
